-- Celebrations Table:

CREATE TABLE celebrations (
    celebration_id INT PRIMARY KEY,
    location VARCHAR(255),
    num_guests INT,
    travel_distance DECIMAL(10, 2),
    celebration_date DATE
);

    post_content TEXT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci,
    FOREIGN KEY (celebration_id) REFERENCES celebrations(celebration_id)
);
